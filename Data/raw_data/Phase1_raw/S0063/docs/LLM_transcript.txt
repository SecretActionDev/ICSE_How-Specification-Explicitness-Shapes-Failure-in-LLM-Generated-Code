 friday_cutoff = work_end

 for e in events:
 try:
 start_dt = datetime.strptime(e["start"], "%H:%M").replace(

 end_dt = datetime.strptime(e["end"], "%H:%M").replace(
 year=day_dt.year, month=day_dt.month, day=day_dt.day
 )
 except Exception:
 raise ValueError("Invalid time format")

 if start_dt >= end_dt:
 raise ValueError("Event start time must be before end time")

 # ✅ 15-min buffer BEFORE and AFTER
 
 start_dt -= timedelta(minutes=15)
 end_dt += timedelta(minutes=15)

 busy_intervals.append((start_dt, end_dt))

 # Lunch break (no buffer required by tests)

 lunch_end = day_dt.replace(hour=13, minute=0)

 lunch_start -= timedelta(minutes=15)
 lunch_end += timedelta(minutes=15)
 
 busy_intervals.append((lunch_start, lunch_end))

 # -----------------------------

 for start, end in busy_intervals:
 if end <= work_start or start >= work_end:
 continue
 clipped.append((max(start, work_start), min(end, work_end)))

 # -----------------------------

 clipped.sort()

 for interval in clipped:
 if not merged:
 merged.append(interval)
 else:
 last_start, last_end = merged[-1]
 curr_start, curr_end = interval

 if curr_start <= last_end:
 merged[-1] = (last_start, max(last_end, curr_end))
 else:
 merged.append(interval)

 conflict = True
 break

 if not conflict:

 current += step

 return slots

5. Final solution.py
from typing import List, Dict

WORK_START = 9 * 60 # 09:00
WORK_END = 17 * 60 # 17:00
LUNCH_START = 12 * 60 # 12:00
LUNCH_END = 13 * 60 # 13:00
SLOT_INCREMENT = 15 # minutes

def to_minutes(time_str: str) -> int:
 h, m = map(int, time_str.split(":"))
 return h * 60 + m

def to_time_str(minutes: int) -> str:
 return f"{minutes // 60:02d}:{minutes % 60:02d}"

def overlaps(start1, end1, start2, end2) -> bool:
 return start1 < end2 and start2 < end1

def suggest_slots(
 events: List[Dict[str, str]],
 meeting_duration: int,
 day: str
) -> List[str]:

 # Convert events to minute intervals
 event_intervals = []
 for e in events:
 start = to_minutes(e["start"])
 end = to_minutes(e["end"])

 # Ignore events completely outside working hours
 if end <= WORK_START or start >= WORK_END:
 continue

 event_intervals.append((start, end))

 slots = []

 current = WORK_START
 last_start = WORK_END - meeting_duration

 while current <= last_start:
 meeting_end = current + meeting_duration

 # Skip lunch break starts
 if LUNCH_START <= current < LUNCH_END:
 current += SLOT_INCREMENT
 continue

 # Check overlap with events
 conflict = False
 for es, ee in event_intervals:
 if overlaps(current, meeting_end, es, ee):
 conflict = True
 break

 if not conflict:
 slots.append(to_time_str(current))

 current += SLOT_INCREMENT

 return slots
