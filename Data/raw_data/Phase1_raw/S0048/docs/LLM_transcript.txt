from typing import List, Dict

def suggest_slots(
 events: List[Dict[str, str]],
 meeting_duration: int,
 day: str
) -> List[str]:
 """
 Suggest possible meeting start times for a given day.

 Args:
 events: List of dicts with keys {"start": "HH:MM", "end": "HH:MM"}
 meeting_duration: Desired meeting length in minutes
 day: Three-letter day abbreviation (e.g., "Mon", "Tue", ... "Fri")

 Returns:
 List of valid start times as "HH:MM" sorted ascending
 """

 # --- Helper functions ---
 def to_minutes(t: str) -> int:
 h, m = map(int, t.split(":"))
 return h * 60 + m

 def to_time_str(minutes: int) -> str:
 return f"{minutes // 60:02d}:{minutes % 60:02d}"

 # --- Working hours (assumption) ---
 WORK_START = to_minutes("09:00")
 WORK_END = to_minutes("17:00")

 # --- Convert events to minute intervals ---
 busy_intervals = []
 for e in events:
 start = to_minutes(e["start"])
 end = to_minutes(e["end"])
 busy_intervals.append((start, end))

 # Sort events by start time
 busy_intervals.sort()

 # --- Merge overlapping busy intervals ---
 merged = []
 for interval in busy_intervals:
 if not merged or interval[0] > merged[-1][1]:
 merged.append(list(interval))
 else:
 merged[-1][1] = max(merged[-1][1], interval[1])

 # --- Find free intervals ---
 free_intervals = []
 current = WORK_START

 for start, end in merged:
 if start > current:
 free_intervals.append((current, start))
 current = max(current, end)

 if current < WORK_END:
 free_intervals.append((current, WORK_END))

 # --- Generate valid meeting start times ---
 valid_starts = []

 for start, end in free_intervals:
 t = start
 while t + meeting_duration <= end:
 valid_starts.append(to_time_str(t))
 t += 1 # minute granularity

 return valid_starts

from typing import List, Dict

def suggest_slots(events: List[Dict[str, str]], meeting_duration: int, day: str) -> List[str]:

 def to_minutes(t: str) -> int:
 h, m = map(int, t.split(":"))
 return h * 60 + m

 def to_time_str(m: int) -> str:
 return f"{m // 60:02d}:{m % 60:02d}"

 WORK_START = to_minutes("09:00")
 WORK_END = to_minutes("17:00")
 LUNCH_START = to_minutes("12:00")
 LUNCH_END = to_minutes("13:00")
 BUFFER = 15
 SLOT_STEP = 15

 # Convert events to minutes and sort
 busy = [(to_minutes(e["start"]), to_minutes(e["end"])) for e in events]
 busy.sort()

 # Merge overlapping events
 merged = []
 for s, e in busy:
 if not merged or s > merged[-1][1]:
 merged.append([s, e])
 else:
 merged[-1][1] = max(merged[-1][1], e)

 # Build free intervals (without buffer here)
 free = []
 current = WORK_START
 for s, e in merged:
 if s > current:
 free.append((current, s))
 current = max(current, e) # buffer handled when checking slots
 if current < WORK_END:
 free.append((current, WORK_END))

 slots = []
 for start, end in free:
 t = ((start + SLOT_STEP - 1) // SLOT_STEP) * SLOT_STEP # align to 15-min
 while t + meeting_duration <= end:
 # Check if this slot overlaps **any event including buffer**
 conflict = False

 t += SLOT_STEP

 return slots
