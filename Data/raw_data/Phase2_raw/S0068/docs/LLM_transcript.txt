from typing import List, Dict

def is_allocation_feasible(requests: List[Dict[str, int]], resources: Dict[str, int]) -> bool:
 """
 Determines whether a set of resource requests can be satisfied given limited capacity.

 Args:
 requests: A list of dictionaries, where each dictionary represents a request
 (e.g., {"cpu": 2, "memory": 4}).
 resources: A dictionary representing available resources
 (e.g., {"cpu": 10, "memory": 32}).

 Returns:
 True if all requests can be satisfied within the available resources, False otherwise.
 """
 # Create a dictionary to track total usage for each resource type
 total_usage = {key: 0 for key in resources}

 for req in requests:
 for resource_type, amount in req.items():
 # If a request asks for a resource that doesn't exist, it's infeasible
 # (Alternatively, you could treat unknown resources as a separate failure condition)
 if resource_type not in resources:
 return False
 
 total_usage[resource_type] += amount

 # Check if total usage exceeds available capacity for any resource
 for resource_type, limit in resources.items():
 if total_usage[resource_type] > limit:
 return False

 return True
