```python
from typing import Dict, List, Union

Number = Union[int, float]

def is_allocation_feasible(
 resources: Dict[str, Number],
 requests: List[Dict[str, Number]]
) -> bool:
 """
 Determine whether a set of resource requests can be satisfied given limited capacities.

 NEW REQUIREMENT (Lab 6):
 The allocation is feasible **only if**:
 1. No resource is over-allocated (total demand <= capacity for every resource), AND
 2. At least one resource has remaining capacity > 0 after all requests are satisfied.

 Allocations that exactly consume every single resource are now INVALID, even if they
 previously passed the old capacity checks.

 Args:
 resources: Mapping from resource name to total available capacity.
 requests: List of individual requests. Each request is a dict of resource → amount.

 Returns:
 True if the allocation satisfies both the capacity constraints AND leaves
 at least one resource unallocated. False otherwise.

 Raises:
 ValueError: for malformed input (non-dicts, unknown resources, negative values, etc.)
 """
 # === Existing robust error checking (kept from donor) ===
 if not isinstance(resources, dict):
 raise ValueError("Resources must be a dictionary!")

 total_demand: Dict[str, float] = {}

 for i, req in enumerate(requests, 1):
 if not isinstance(req, dict):
 raise ValueError(f"Request #{i} is not a dictionary")

 for res_name, amount in req.items():
 if res_name not in resources:
 return False # unknown resource → infeasible

 try:
 amount = float(amount)
 except (TypeError, ValueError):
 raise ValueError(
 f"Invalid numeric value for resource '{res_name}' "
 f"in request #{i}: {amount!r}"
 )

 if amount < 0:
 raise ValueError(
 f"Negative value not allowed for resource '{res_name}' "
 f"in request #{i}: {amount}"
 )

 total_demand[res_name] = total_demand.get(res_name, 0.0) + amount

 # === Capacity check (old behavior) ===
 for res_name, demanded in total_demand.items():
 available = float(resources[res_name])
 if demanded > available:
 return False

 # === NEW REQUIREMENT: at least one resource must remain unallocated ===
 if resources: # only enforce when resources exist
 has_remaining = any(
 total_demand.get(res, 0.0) < float(avail)
 for res, avail in resources.items()
 )
 if not has_remaining:
 return False # everything fully consumed → invalid

 return True
```
