* **`resources`** → total capacity

We must ensure that **for every resource type**:


## ✅ Final Implementation

```python
from typing import Dict, List, Union

Number = Union[int, float]

def is_allocation_feasible(
 resources: Dict[str, Number],
 requests: List[Dict[str, Number]]
) -> bool:
 """
 Determine whether a set of resource requests can be satisfied given limited capacities.
 """

 # Validate resource capacities
 for capacity in resources.values():
 if capacity < 0:
 return False

 total_requested: Dict[str, Number] = {}

 # Sum all requests
 for request in requests:
 for resource, amount in request.items():
 # No negative requests
 if amount < 0:
 return False

 total_requested[resource] = total_requested.get(resource, 0) + amount

 # Compare totals with available resources
 for resource, total in total_requested.items():
 # Resource must exist
 if resource not in resources:
 return False

 # Must not exceed capacity
 if total > resources[resource]:
 return False

 return True
```
