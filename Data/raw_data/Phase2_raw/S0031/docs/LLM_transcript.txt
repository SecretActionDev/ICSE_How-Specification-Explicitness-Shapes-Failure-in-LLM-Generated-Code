from typing import Dict, List, Union
import math

Number = Union[int, float]

def is_allocation_feasible(
 resources: Dict[str, Number],
 requests: List[Dict[str, Number]]
) -> bool:
 """
 Determines whether a set of resource requests can be feasibly allocated.

 An allocation is feasible if:
 1. All requested resources exist
 2. No resource exceeds its capacity
 3. All inputs are valid (non-negative, finite numbers)
 4. At least one resource remains partially unallocated after assignment
 """

 # Validate resources
 if not isinstance(resources, dict):
 raise ValueError("Resources must be a dictionary")

 for name, cap in resources.items():
 if not isinstance(cap, (int, float)):
 raise ValueError("Invalid capacity type")
 if math.isnan(cap) or math.isinf(cap) or cap < 0:
 raise ValueError("Invalid capacity value")

 # Track total usage
 usage = {r: 0 for r in resources}

 # Process each request
 for req in requests:
 if not isinstance(req, dict):
 raise ValueError("Each request must be a dictionary")

 for r, amount in req.items():

 # Unknown resource → infeasible
 if r not in resources:
 return False

 if not isinstance(amount, (int, float)):
 raise ValueError("Invalid request amount type")

 if math.isnan(amount) or math.isinf(amount) or amount < 0:
 raise ValueError("Invalid request amount value")

 usage[r] += amount

 if usage[r] > resources[r]:
 return False

 # New requirement:
 # At least one resource must have remaining capacity
 for r in resources:
 if usage[r] < resources[r]:
 return True

 # All resources fully consumed
 return False

An invariant is a property that must hold for every reachable state during execution.

Your new rule:

Applies only to final / completed allocations

Does not need to hold during intermediate accumulation of requests

Is checked once, at the end

✅ Invariant example (must always hold)
For all resources r:
 usage[r] ≤ capacity[r]

Holds after every request

There exists a resource r such that:

Only meaningful after allocation is complete

Does NOT have to hold mid-execution
