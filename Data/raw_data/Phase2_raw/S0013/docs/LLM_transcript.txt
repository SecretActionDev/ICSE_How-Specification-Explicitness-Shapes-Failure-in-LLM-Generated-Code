from collections import Counter
from typing import Dict, List
from numbers import Number

def is_allocation_feasible(
 resources: Dict[str, Number],
 requests: List[Dict[str, Number]]
) -> bool:

 total_requests: Counter[str] = Counter()

 # Step 1: Accumulate the total requests
 for req in requests:
 if not isinstance(req, dict):
 raise ValueError(f"Expected dict in requests, got {type(req)}: {req}")
 for res, count in req.items():
 total_requests[res] += count

 # Step 2: Check if any resource exceeds its available quantity
 for res, total in total_requests.items():
 if total > resources.get(res, 0):
 return False # A request exceeds available resources

 # Step 3: Ensure that at least one resource remains unallocated
 for res, available in resources.items():
 allocated = total_requests.get(res, 0)
 if available - allocated > 0:
 return True # Found at least one unallocated resource

 return False # No resources remain unallocated
